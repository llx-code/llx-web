随机迷宫生成核心算法 (DFS实现) 
// 基于DFS的随机迷宫生成核心算法
// 时间复杂度: O(rows * cols)
// 空间复杂度: O(rows * cols)

#include <iostream>
#include <vector>
#include <stack>
#include <cstdlib>
#include <ctime>
using namespace std;

// 迷宫尺寸
const int rows = 15;
const int cols = 30;

// 迷宫地图，true表示墙，false表示通路
bool maze[rows][cols];
// 记录是否访问过
bool visited[rows][cols];

// 方向数组：上、右、下、左
const int dx[] = {-1, 0, 1, 0};
const int dy[] = {0, 1, 0, -1};

// 生成迷宫核心算法
void generateMaze() {
    srand(time(0));
    
    // 栈用于DFS
    stack<pair<int, int>> s;
    
    // 从左上角开始(1,1)，保证是奇数坐标
    int startX = 1, startY = 1;
    s.push({startX, startY});
    visited[startX][startY] = true;
    maze[startX][startY] = false;  // 起点设为通路
    
    // 方向顺序数组，用于随机打乱
    int dirs[4] = {0, 1, 2, 3};
    
    while (!s.empty()) {
        auto current = s.top();
        int x = current.first;
        int y = current.second;
        
        // 随机打乱方向
        for (int i = 0; i < 4; i++) {
            swap(dirs[i], dirs[rand() % 4]);
        }
        
        // 检查所有方向
        bool hasUnvisited = false;
        for (int i = 0; i < 4; i++) {
            int dir = dirs[i];
            int nx = x + 2 * dx[dir];  // 跳过一个格子
            int ny = y + 2 * dy[dir];
            
            // 检查是否在范围内且未访问
            if (nx > 0 && nx < rows-1 && ny > 0 && ny < cols-1 && !visited[nx][ny]) {
                // 标记为已访问并设为通路
                visited[nx][ny] = true;
                maze[nx][ny] = false;
                
                // 打通当前格子与下一个格子之间的墙
                maze[x + dx[dir]][y + dy[dir]] = false;
                
                // 入栈继续探索
                s.push({nx, ny});
                hasUnvisited = true;
                break;
            }
        }
        
        // 如果没有未访问的邻居，则回溯
        if (!hasUnvisited) {
            s.pop();
        }
    }
    
    // 设置入口和出口
    maze[startX][startY-1] = false;  // 入口
    maze[rows-2][cols-1] = false;    // 出口
}

// 算法核心说明：
// 1. 使用深度优先搜索(DFS)思想，通过栈实现
// 2. 从起点开始，随机选择一个未访问的方向
// 3. 移动两步(跳过一个格子)，并打通中间的墙
// 4. 当无法继续前进时，回溯到上一个格子
// 5. 重复过程直到所有可达格子都被访问
// 这种方法能生成连通性好、路径复杂的迷宫

汉诺塔递归核心算法实现 
// 汉诺塔递归核心算法
// 将n个盘子从a柱移动到c柱，借助b柱
// 时间复杂度: O(2^n)
// 空间复杂度: O(n) - 递归栈深度

#include <iostream>
using namespace std;

// 移动一个盘子的操作
void move(char from, char to) {
    cout << "将盘子从 " << from << " 移动到 " << to << endl;
}

// 汉诺塔递归求解函数
// 参数说明：
// n: 盘子数量
// a: 源柱子
// b: 辅助柱子
// c: 目标柱子
void hanoi(int n, char a, char b, char c) {
    // 递归终止条件：只有一个盘子时直接移动
    if (n == 1) {
        move(a, c);
        return;
    }
    
    // 递归步骤1：将n-1个盘子从a柱移到b柱，借助c柱
    hanoi(n - 1, a, c, b);
    
    // 步骤2：将第n个盘子从a柱直接移到c柱
    move(a, c);
    
    // 递归步骤3：将n-1个盘子从b柱移到c柱，借助a柱
    hanoi(n - 1, b, a, c);
}

// 算法核心说明：
// 1. 采用分治思想，将复杂问题分解为简单子问题
// 2. 递归终止条件：1个盘子直接移动
// 3. 递归步骤：
//    - 先将n-1个盘子从源柱移到辅助柱
//    - 将第n个盘子从源柱移到目标柱
//    - 最后将n-1个盘子从辅助柱移到目标柱
// 4. 移动次数：2^n - 1，是最优解法
// 该算法清晰展示了递归的精髓：将大问题分解为相似的小问题
